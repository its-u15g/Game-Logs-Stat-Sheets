function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // 1行目のヘッダーを取得（ここで設定した列名と、送信されたデータ名をマッチングさせます） const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; // アプリから送信されたデータを読み込む const data = JSON.parse(e.postData.contents); const payload = Array.isArray(data) ? data : [data]; const timestamp = new Date(); // 各プレイヤーのデータを処理 payload.forEach(playerData => { let rowData = new Array(headers.length).fill(''); // タイムスタンプをセット const timeIndex = headers.indexOf('タイムスタンプ'); if (timeIndex !== -1) rowData[timeIndex] = timestamp; // 送信されたデータのキー名と、スプレッドシートのヘッダー名が一致する列に値を入れる for (const key in playerData) { const colIndex = headers.indexOf(key); if (colIndex !== -1) { rowData[colIndex] = playerData[key]; } } // CP（コートプレイヤー）の場合、合計ゴール数と合計シュート数を自動計算して追加 if (playerData.position === 'CP') { const totalGoals = ['L_goals', 'M_goals', 'C_goals', 'P_goals', 'F_goals', 'S_goals', '7m_goals'] .reduce((sum, k) => sum + (playerData[k] || 0), 0); const totalMisses = ['L_miss', 'M_miss', 'C_miss', 'P_miss', 'F_miss', 'S_miss', '7m_miss'] .reduce((sum, k) => sum + (playerData[k] || 0), 0); const goalsIndex = headers.indexOf('total_goals'); const shotsIndex = headers.indexOf('total_shots'); if (goalsIndex !== -1) rowData[goalsIndex] = totalGoals; if (shotsIndex !== -1) rowData[shotsIndex] = totalGoals + totalMisses; } // シートに1行追加 sheet.appendRow(rowData); }); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "message": "Data saved successfully" })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": error.message })) .setMimeType(ContentService.MimeType.JSON); } }
